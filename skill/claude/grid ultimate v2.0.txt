technical_grid_specification:

  use_when: "User provides image without 'cinema' keyword and without text mode triggers"

  grid_layout: "3x3 (9 panels total)"

  output_format: "Single master image with all 9 panels in grid arrangement"

  

  row_1_establishing_context:

    panel_1:

      shot_type: "Extreme Long Shot (ELS)"

      label: "ELS"

      description: "Subject(s) appear small within vast environment"

      framing: "Maximum context, showing full location/setting"

      purpose: "Establish scale and environment"

    

    panel_2:

      shot_type: "Long Shot (LS)"

      label: "LONG SHOT"

      description: "Complete subject(s) visible head-to-toe or full object"

      framing: "Subject takes 40-60% of frame, environment still visible"

      purpose: "Show full body/object with spatial context"

    

    panel_3:

      shot_type: "Medium Long Shot (American/3-4)"

      label: "MED LONG"

      description: "For people: knees up. For objects: 3/4 view"

      framing: "Subject takes 60-75% of frame"

      purpose: "Balance between detail and context"

  

  row_2_core_coverage:

    panel_4:

      shot_type: "Medium Shot (MS)"

      label: "MEDIUM SHOT"

      description: "For people: waist up. For objects: central portion"

      framing: "Subject fills 70-80% of frame"

      purpose: "Focus on action/interaction, conversational framing"

    

    panel_5:

      shot_type: "Medium Close-Up (MCU)"

      label: "MED CLOSE"

      description: "For people: chest up. For objects: front/main feature"

      framing: "Subject fills 80-90% of frame"

      purpose: "Intimate framing, capture expression/detail"

    

    panel_6:

      shot_type: "Close-Up (CU)"

      label: "CLOSE-UP"

      description: "For people: face fills frame. For objects: key identifying feature"

      framing: "Subject fills 90-95% of frame"

      purpose: "Maximum emotional connection or detail focus"

  

  row_3_details_and_angles:

    panel_7:

      shot_type: "Extreme Close-Up (ECU)"

      label: "EXTREME CU"

      description: "Macro detail: eyes, hands, texture, logo, surface quality"

      framing: "Single detail fills entire frame"

      purpose: "Hyperdetail, sensory experience"

    

    panel_8:

      shot_type: "Low Angle (Worm's Eye)"

      label: "LOW ANGLE"

      description: "Camera positioned at ground level looking up at subject"

      framing: "Subject appears imposing/heroic, sky/ceiling visible"

      purpose: "Power dynamic, dramatic perspective"

    

    panel_9:

      shot_type: "High Angle (Bird's Eye)"

      label: "HIGH ANGLE"

      description: "Camera positioned above looking down at subject"

      framing: "Top-down or steep angle, ground/floor visible"

      purpose: "Vulnerability or overview perspective"

  

  technical_requirements:

    consistency_checklist:

      - "Same subjects in all 9 panels (no additions or removals)"

      - "Same clothing/wardrobe across all shots"

      - "Same environment/location throughout"

      - "Same lighting setup (time of day, quality, direction)"

      - "Style medium maintained (photo stays photo, art stays art)"

      - "Depth of field adjusts realistically (wider shots deeper focus, close-ups shallower)"

    

    labeling_requirements:

      position: "Top-left corner of each panel"

      color: "White text with subtle shadow for legibility"

      font: "Clean sans-serif, bold weight"

      size: "Readable at thumbnail scale"

    

    grid_composition:

      aspect_ratio: "Each panel should maintain consistent aspect (typically 16:9 or 4:3)"

      spacing: "Thin consistent borders between panels"

      overall_layout: "Professional contact sheet aesthetic"

# NARRATIVE CINEMA MODE (Storytelling Sequence)

# ============================================================


narrative_cinema_specification:

  use_when: "User input contains keywords: cinema, Cinema, CINEMA, storyboard, cinematic"

  keyframe_count: "9-12 frames (flexible based on story needs)"

  output_format: "Single master contact sheet image + full text breakdown"

  

  step_1_analysis:

    label: "🎬 SCENE & STYLE BREAKDOWN"

    actions:

      detect_style:

        question: "Is this PHOTOREALISTIC or STYLIZED?"

        photorealistic_indicators: "Photography, CGI render, realistic lighting, natural DoF"

        stylized_indicators: "2D animation, digital painting, sketch, anime, cel-shading, flat colors"

        critical_rule: "STRICTLY maintain the detected style - do NOT force photorealism on stylized art"

      

      identify_subjects:

        action: "List ALL key subjects present in reference"

        include: "People (count, appearance), objects, vehicles, animals"

        spatial_relationships: "Describe how subjects relate spatially and interact"

      

      analyze_environment:

        components: "Location type, spatial layout, time of day, weather conditions"

        lighting_quality: "Hard/soft, direction, color temperature"

        atmosphere: "Mood conveyed by environment"

      

      define_visual_anchors:

        definition: "3-6 traits that MUST stay constant across all frames"

        examples: "Color palette, specific props, architectural elements, costume details, line weight/grain"

  

  step_2_theme_and_story:

    label: "📖 NARRATIVE ARCHITECTURE"

    components:

      theme:

        definition: "One sentence capturing the core idea"

        example: "Solitude in urban sprawl" or "Defiant hope against decay"

      

      logline:

        definition: "One restrained trailer-style sentence"

        format: "[Subject] [action/situation] [emotional context]"

        example: "A lone traveler navigates an abandoned cityscape searching for signs of life"

      

      emotional_arc:

        beat_1_setup:

          timing: "0-3 seconds"

          purpose: "Establish normal state, introduce subject"

          example: "Character in familiar environment, calm before change"

        

        beat_2_build:

          timing: "3-8 seconds"

          purpose: "Introduce tension, movement, change"

          example: "Subject begins journey, obstacles appear, stakes rise"

        

        beat_3_turn:

          timing: "8-15 seconds"

          purpose: "Peak tension, revelation, confrontation"

          example: "Discovery moment, emotional climax, visual surprise"

        

        beat_4_payoff:

          timing: "15-20 seconds"

          purpose: "Resolution or cliffhanger, emotional landing"

          example: "Subject transformed, question answered or deepened"

  

  step_3_cinematic_approach:

    label: "🎥 FILMMAKING STRATEGY"

    components:

      shot_progression:

        photorealistic_strategy: "Wide → Medium → Close → Detail, with motivated camera moves"

        stylized_strategy: "Pan → Parallax → Zoom simulation → Cut to detail"

        logic: "Build intimacy gradually OR shock with contrast (wide to ECU)"

      

      camera_movement:

        photorealistic_options: "Dolly, crane, gimbal, handheld, static, zoom"

        stylized_options: "Multiplane parallax, camera pan, simulated zoom, cut transitions"

        purpose: "Match movement to emotional beat (smooth = calm, shake = tension)"

      

      lens_and_aesthetics:

        if_photorealistic:

          specify: "Focal length (24mm, 50mm, 85mm, etc.), aperture feel (shallow/deep DoF)"

          shutter_feel: "24fps cinematic, 60fps smooth, 120fps slow-mo"

        

        if_stylized:

          specify: "Line weight consistency, brush stroke flow, cel-shading intensity"

          perspective: "Fixed orthographic, exaggerated foreshortening, isometric"

      

      light_and_color:

        contrast: "Low-key (dramatic shadows) vs high-key (bright, even)"

        palette: "Warm, cool, monochrome, saturated, desaturated"

        consistency: "Match reference image's color science exactly"

  

  step_4_keyframe_construction:

    label: "🎞️ KEYFRAME SEQUENCE"

    default_count: "9-12 frames"

    

    mandatory_shots:

      - "1 environment-establishing wide shot"

      - "1 intimate close-up (face or key subject detail)"

      - "1 extreme close-up (macro detail)"

      - "1 power-angle shot (low or high angle for drama)"

    

    keyframe_format:

      template: |

        [KF# | duration (sec) | shot type]

        - Composition: subject placement, rule of thirds, leading lines

        - Action/Beat: what visibly happens, story moment

        - Camera: movement type, speed, direction

        - Style/Render: texture instruction (photoreal focus OR flat shading, line art, etc)

        - Lighting & Grade: maintain consistency with reference

        - Sound/Atmosphere (optional): ambient sound description

      

      example: |

        [KF1 | 2s | EXTREME LONG SHOT]

        - Composition: Character small in lower right third, vast cityscape dominates

        - Action/Beat: Character stands at edge of rooftop, surveying empty streets below

        - Camera: Static, wide lens (24mm equivalent), deep focus

        - Style/Render: Photorealistic with natural DoF, slight film grain

        - Lighting & Grade: Golden hour backlight, warm 3200K, high contrast

        - Sound/Atmosphere: Wind ambience, distant city hum

  

  step_5_contact_sheet_output:

    label: "📋 MASTER CONTACT SHEET"

    requirements:

      single_master_image: "ONE composite image containing all keyframes as separate panels"

      grid_layout: "Typically 3x3 (9 frames) or 3x4 (12 frames) depending on story needs"

      panel_labeling: "Each panel labeled with KF number + shot type (e.g., 'KF3 - CLOSE-UP')"

      style_adherence: "Grid must look like it belongs to exact same universe as reference"

      consistency: "Same subjects, wardrobe, environment across all panels"

    

    after_image_output:

      provide: "Full text breakdown of all keyframes (as formatted in step 4)"

      include: "Scene analysis, theme, story arc, cinematic approach"

    rule: "Clothing, accessories, and styling CANNOT change between panels"

    violations: "Moving from interior to exterior, changing weather, altering furniture"

    enforcement: "All panels must be frameable within the same physical space"

  

  lighting_consistency:

    rule: "Light direction, quality, and color temperature stay constant"

    acceptable_variation: "Depth of field can change based on shot distance"

    violations: "Changing time of day, moving light sources, altering shadow direction"

    enforcement: "Reference lighting analysis for each panel generation"

  

  style_medium_lock:

    rule: "If reference is photorealistic, ALL panels are photorealistic. If stylized, ALL match art style."

    violations: "Mixing photo and illustration, adding realistic textures to cartoons"

    enforcement: "Style consistency check before each panel generation"

    close_ups: "Shallower focus (f/1.4-f/2.8), background bokeh"
