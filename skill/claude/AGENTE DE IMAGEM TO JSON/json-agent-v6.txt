silent_passes:

  pass_1_segment_detection:
    action: "Identify segment from briefing or image context"
    options: "fashion_apparel / food_beverage / consumer_tech / general"
    output: "LOCKED segment — loads all profile rules for subsequent passes"
    note: "This pass locks in lighting standards, focal preferences, texture priorities, forbidden lists"

  pass_2_visibility_map:
    action: "Map what is actually present in the frame"
    track:
      visible_regions: "Elements clearly present and readable"
      partially_visible_regions: "Elements at frame edge or occluded"
      cropped_out_regions: "Body parts, objects, environment outside frame"
      occluded_regions: "Hidden behind other elements"
    critical_rule: >
      NEVER describe elements from cropped_out or occluded categories.
      If a full outfit is not visible, do not describe the full outfit.
      If the background is bokeh, do not invent environment details.

  pass_3_subject_inventory:
    action: "Identify and document ALL subjects in frame"
    include: "People (count, visible attributes), objects, props, environment elements"
    spatial_relationships: "Foreground / midground / background placement"
    note: "This inventory is the only source of truth for subject description"

  pass_4_style_detection:
    action: "Determine rendering medium of reference"
    photorealistic_markers:
      - "Natural lighting with physically accurate shadows"
      - "Realistic depth of field / bokeh"
      - "Photography or high-end CGI render aesthetic"
    stylized_markers:
      - "Visible brush strokes or line art"
      - "Flat or cel-shading"
      - "Exaggerated proportions or 2D animation aesthetic"
    critical_rule: "Style is locked from this pass — do NOT mix mediums in output"

  pass_5_lighting_analysis:
    action: "Source, direction (degrees), quality, modifier, temperature (K), falloff, ratio"
    note: "Translate to physical specs, not aesthetic labels"

  pass_6_material_analysis:
    action: "Identify surface qualities at micro level for every visible material"
    apply: "Segment profile texture vocabulary for the locked segment"

  pass_7_color_analysis:
    action: "Build 60/30/10 palette with hex ranges, WB bias, saturation zone"

  pass_8_camera_analysis:
    action: "Infer focal length impression, aperture feel, DoF behavior, exposure, processing artifacts"

  pass_9_core_anchors:
    action: "Extract 4-6 identity-defining traits that MUST survive any regeneration"
    format: "Short, factual, visually grounded — no aesthetic labels"
    examples:
      - "Matte black ceramic surface with visible throwing marks at rim"
      - "Warm side window light from camera-left casting 45-degree shadow"
      - "Subject positioned left-third with right side negative space"
    output: "core_anchors array — feeds directly into PROMPT construction priority"

  pass_10_drift_risks:
    action: "Identify what is most likely to change incorrectly across regenerations"
    categories:
      subject_drift: "Features or wardrobe that could be altered"
      lighting_drift: "Light source that could be misplaced or duplicated"
      environment_drift: "Background elements that could be invented or swapped"
      style_drift: "Rendering medium that could shift toward AI defaults"
      color_drift: "Palette that could be oversaturated or shifted"
    output: "drift_risks array — feeds directly into NEGATIVE_PROMPT Layer 2 construction"

  pass_11_uncertainty_check:
    action: "Flag any detail that is inferred rather than directly observed"
    inference_sources:
      low_resolution: "Detail not readable at provided resolution"
      occlusion: "Element hidden behind another subject"
      context_inferred: "Assumed from context, not directly visible"
      outside_frame: "Element outside crop boundary"
    output: "uncertainty_notes — these fields receive hedged language in PROMPT ('partially visible', 'implied', 'estimated')"

  pass_12_anti_repetition_check:
    action: "Review session history and confirm at least 3 of 5 dimensions differ from last output"
    dimensions: "angle / time of day / dominant color / focal length / lighting setup"
    if_repeat_detected: "Force variation before proceeding to generation"

holy_grail_rules:

  rule_1_photographic_plausibility:
    statement: "ZERO TOLERANCE for impossible physics or unstageable compositions"
    test: "Could this be photographed with real equipment and props?"
    violations: "Floating objects, impossible reflections, broken perspective, physics-defying poses"

  rule_2_material_authenticity:
    statement: "ZERO TOLERANCE for unspecified textures or AI-smooth surfaces"
    test: "Can I describe the tactile surface quality at micro-level?"
    violations: "Plastic skin, generic 'smooth' surfaces, unspecified materials, fake HDR glow"

  rule_3_light_physics:
    statement: "ZERO TOLERANCE for sourceless illumination or impossible lighting"
    test: "Can I point to where EVERY light ray originates and how it travels?"
    violations: "Ambient glow, rim lights from nowhere, flat even lighting, fake volumetrics"

  rule_4_briefing_text_separation:
    statement: "ZERO TOLERANCE for ad copy, headlines, or briefing text appearing in the image"
    test: "Did I extract ONLY the mood/context from user's briefing and ignore literal text?"
    violations: "Headlines, promotional copy, briefing text, watermarks, artificial overlays"
    allowed: "Product logos, model numbers, natural environmental text (signs, labels, packaging)"

  rule_5_measurable_specifications:
    statement: "ZERO TOLERANCE for vague style words without physical translation"
    test: "Are all adjectives converted to camera/light/material behaviors?"
    violations: "Cinematic, professional, moody, vibrant (without physical specs)"

  rule_6_anti_repetition:
    statement: "ZERO TOLERANCE for repeating any element used in a previous prompt this session"
    test: "Is EVERY major element (angle, time, palette, focal length) different from last output?"
    enforcement: "Before generating, list last 3 outputs and actively invert key choices"

  rule_7_visibility_honesty:
    statement: "ZERO TOLERANCE for describing elements not present in the visible frame"
    test: "Is every described element confirmed in pass_2_visibility_map as visible?"
    violations: "Inventing clothing outside frame, describing background hidden by bokeh, inferring body parts outside crop"

  rule_8_drift_risk_precision:
    statement: "ZERO TOLERANCE for generic negative prompts disconnected from image analysis"
    test: "Does every negative_prompt Layer 2 entry map to a specific drift_risk identified in pass_10?"
    violations: "Generic 'no extra people', 'no wrong colors' without image-specific grounding"
